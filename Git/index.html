    Git 使用

    配置级别为三个

    --local  # 本地基别 优先级最高 针对当前仓库
--global # 全局级别 优先级没有本地高 针对当前用户
--system # 系统基别 优先级最低

    最基本的配置

    git config --global user.name 用户名 # 如果有值 则为修改没有则为添加
git config --global user.email 你的email地址
git config --global --add user.name 你要添加的用户名 # 添加新的值
# 查询配置的值
git config --get user.name
git config user.name
# 查看某个配置级别下的所有配置
git config --list --配置级别
# 删除配置
git config --配置基本 --unset 配置项 表达式
git config --global --unset user.name ChaoHer
# 如果只有一个值 可以直接使用
git config --global --unset user.name
# 给子参数起起别名
git config --global alias.co checkout # 给checkout起别名 叫co
git config --global alias.logs "log --oneline"
# 查看git config 文档 有三个命令
git config --help
git help config
man git-config

    常用命令

    git init 仓库名 # 初始化一个git仓库
# 或者进入到你要初始化仓库的目录
git init
git add 你要添加的到暂存区域的文件名 以空格分割 # 添加文件到暂存区
git commit -m "frist commit" # 提交到git库
git status # 查看
# 删除一个文件
git rm 要删除的文件 # 同时删除工作区和暂存区域文件
git rm -rf --cache 删除暂存区但不删除本地
# 还原一个文件
git reset HEAD 文件名 # 还原暂存区域文件
git checkout 文件名 # 还原工作区文件
# 只删除暂存区文件不删除工作区文件
git rm --cached 文件名
# 重命名文件
git mv 文件名 移动到的位置|或者重命名的名称
# 查看提交历史
git log

    设置忽略文件列表

    git add -A # 添加所有文件
# 在git工作区创建 .gitignore 文件
touch .gitignore
vim .gitignore
# 配置如下
*.[oa] # 忽略 以o或者a结尾的文件
*.html # 忽略 .html 结尾的文件
*~ # 忽略vim中间文件 就是vim自动创建的备份文件
!index.html # 单独纳入index.html文件 不忽略 index.html
# 如果需要忽略一个以感叹号开头的文件 注意加 反斜杠 \
# 比如有个文件叫 !chao.md 先要忽略 需要添加 \!chao.md
# 忽略目录 后面加斜杠则代表匹配目录 如果后面不加则匹配目录或文件
**/chao/ # 只匹配目录
**/chao  # 匹配目录或文件
# mac 下注意添加
.DS_Store

    git 分支

    git branch 分支名 # 创建分支
git checkout 分支名 # 切换分支
